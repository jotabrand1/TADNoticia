
# ============================================================
# Makefile — Sistema de Noticias
# Compativel com Linux, WSL e Windows (PowerShell/CMD)
# Uso:
#   make        → compila o projeto
#   make clean  → remove arquivos gerados
#   make run    → compila e executa
# ============================================================
 
CC     = gcc
TARGET = prog
SRCS   = main.c Noticia.c ListaCabCau.c ListaEncadeada.c
OBJS   = $(SRCS:.c=.o)
 
# Detecta o sistema operacional
ifeq ($(OS), Windows_NT)
	RM     = del /Q
	EXEC   = $(TARGET).exe
else
	RM     = rm -f
	EXEC   = ./$(TARGET)
endif
 
# Regra padrão
all: $(TARGET)
 
# Linka os .o e gera o executável
$(TARGET): $(OBJS)
	$(CC) $(OBJS) -o $(TARGET)
 
# Compila cada .c em .o
main.o: main.c Noticia.h ListaCabCau.h ListaEncadeada.h
	$(CC) -c main.c
 
Noticia.o: Noticia.c Noticia.h
	$(CC) -c Noticia.c
 
ListaCabCau.o: ListaCabCau.c ListaCabCau.h Noticia.h
	$(CC) -c ListaCabCau.c
 
ListaEncadeada.o: ListaEncadeada.c ListaEncadeada.h Noticia.h
	$(CC) -c ListaEncadeada.c
 
# Executa o programa
run: $(TARGET)
	$(EXEC)
 
# Remove arquivos gerados
clean:
	$(RM) $(OBJS) $(TARGET) $(TARGET).exe
 
.PHONY: all run clean
